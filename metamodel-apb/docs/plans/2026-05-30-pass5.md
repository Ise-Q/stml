def stationary_bootstrap_cer_diff_ci(
    r_alt, r_base, *, risk_aversion: float = 5.0, alpha: float = 0.05,
    reps: int = 2000, seed: int = 42,
) -> tuple[float, float]:
    """Paired, studentised stationary block-bootstrap CI for CER(r_alt) − CER(r_base).

    r_alt and r_base are the SAME strategy rescaled, so a single block-index draw is applied to
    both (independent resampling would inflate the band and bias the decision toward revert).
    CER(r) = mean(r) − ½·λ·var(r). Studentised with a delta-method (influence-function) SE of the
    paired two-moment difference functional — no nested bootstrap (cf. the Sharpe CI). Returns
    (lo, hi) on the difference; lo > 0 means the gain is bootstrap-confident. Cite Politis–Romano
    (1994); Ledoit–Wolf (2008) for the studentised block bootstrap.
    """
    from arch.bootstrap import StationaryBootstrap, optimal_block_length

    a = np.asarray(r_alt, dtype=float)
    b = np.asarray(r_base, dtype=float)
    mask = np.isfinite(a) & np.isfinite(b)
    a, b = a[mask], b[mask]
    if a.size < 8:
        return (float("nan"), float("nan"))
    data = np.column_stack([a, b])
    block = float(optimal_block_length(a - b)["stationary"].iloc[0])

    def _cer_diff(d):
        c = d.mean(axis=0) - 0.5 * risk_aversion * d.var(axis=0, ddof=1)
        return np.asarray([c[0] - c[1]])

    def _se(theta, d):  # noqa: ARG001 — arch passes (theta, data)
        def psi(x):
            xc = x - x.mean()
            return xc - 0.5 * risk_aversion * (xc**2 - x.var(ddof=0))
        infl = psi(d[:, 0]) - psi(d[:, 1])
        return np.asarray([np.std(infl, ddof=1) / np.sqrt(d.shape[0])])

    bs = StationaryBootstrap(block, data, seed=seed)
    ci = bs.conf_int(_cer_diff, reps=reps, method="studentized",
                     std_err_func=_se, size=1 - alpha)
    return (float(ci[0, 0]), float(ci[1, 0]))
```

def _sleeve(scale, gamma_sign, n=400, seed=0):
    """A synthetic sleeve with pnl=side·mkt and a chosen convexity sign at a given return scale."""
    rng = np.random.default_rng(seed)
    mkt = rng.normal(0.0, scale, size=n)
    side = np.ones(n)
    pnl = side * mkt + gamma_sign * (mkt**2)   # planted convexity
    return mkt, pnl

    small = _sleeve(0.01, -1.0, seed=2)    # small-scale sleeve

    # internal standardised arrays exposed for the invariant check

def standardise_then_repool_tm(sleeves, *, _return_debug: bool = False):
    """S5.12 (LR-9 rec #5): vol-target each sleeve to a COMMON scale — divide BOTH mkt and pnl by
    that sleeve's mkt return-std (preserving pnl=side·mkt) — then pool and re-estimate the Treynor–
    Mazuy γ. Compare to the trade-count-weighted average of per-sleeve γ. Collapse of the positive
    pooled γ toward the (negative) weighted average is in-data proof the +1.18 is an aggregation
    artefact, not timing skill. `sleeves` = list of (label, mkt, pnl, gamma_sleeve, n)."""
    mkts, pnls, gammas, ns, dbg = [], [], [], [], []
    for _label, mkt, pnl, g_sleeve, n in sleeves:
        s = float(np.std(mkt, ddof=1))
        if not np.isfinite(s) or s == 0.0 or len(mkt) < 4:
            continue
        m_std, p_std = mkt / s, pnl / s          # SAME factor → preserves pnl=side·mkt
        mkts.append(m_std)
        pnls.append(p_std)
        gammas.append(g_sleeve)
        ns.append(n)
        dbg.append((m_std, p_std, np.sign(np.where(mkt == 0, 1, mkt) * 0 + 1)))
    g_std, t_std, p_std = treynor_mazuy(np.concatenate(mkts), np.concatenate(pnls))
    g_wavg = float(np.average(gammas, weights=np.asarray(ns, dtype=float)))
    if _return_debug:
        return g_std, t_std, p_std, g_wavg, {"per_sleeve": dbg}
    return g_std, t_std, p_std, g_wavg
```
